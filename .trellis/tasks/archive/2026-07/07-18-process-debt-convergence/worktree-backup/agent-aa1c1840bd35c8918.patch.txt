+/// Extract a digest from Docker image inspect JSON.
+///
+/// Prefers `RepoDigests[0]` (returns the part after `@`),
+/// falls back to `Id`, then `"unknown"`.
+fn extract_digest(inspect: &serde_json::Value) -> String {
+    if let Some(digests) = inspect.get("RepoDigests").and_then(|v| v.as_array())
+        && let Some(first) = digests.first().and_then(|v| v.as_str())
+    {
+        if let Some(digest) = first.split('@').nth(1) {
+            return digest.to_string();
+        }
+        return first.to_string();
+    }
+    inspect
+        .get("Id")
+        .and_then(|v| v.as_str())
+        .unwrap_or("unknown")
+        .to_string()
+}
+
+/// Extract the image ID from Docker container inspect JSON's `.Image` field.
+fn extract_container_image_id(inspect: &serde_json::Value) -> String {
+    inspect
+        .get("Image")
+        .and_then(|v| v.as_str())
+        .unwrap_or("unknown")
+        .to_string()
+}
